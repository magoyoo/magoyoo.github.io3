func floydWarshall(_ graph: [[Int]], _ numNodes: Int) -> [[Int]] {
    var dist = graph

    for k in 0..<numNodes {
        for i in 0..<numNodes {
            for j in 0..<numNodes {
                if i == j || i == k || j == k {
                    continue
                }

                if dist[i][k] == Int.max || dist[k][j] == Int.max {
                    continue
                }

                if dist[i][k] + dist[k][j] < dist[i][j] {
                    dist[i][j] = dist[i][k] + dist[k][j]
                }
            }
        }
    }

    return dist
}
